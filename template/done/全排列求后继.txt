bool next(vector<int> &now)
{
    int cnt = now.size(), a = -1, b, t = INT_MAX;
    for (int i = 0; i + 1 < cnt; ++i)
        if (now[i] < now[i + 1]) a = i;
    if (a == -1)
    {
        return false;
    }
    for (int i = a + 1; i < cnt; ++i)
        if ((now[i] > now[a]) && (now[i] < t))
        {
            t = now[i];
            b = i;
        }
    swap(now[a], now[b]);
    for (int i = a + 1; a + cnt - i > i; ++i)
        swap(now[i], now[a + cnt - i]);
    return true;
}
