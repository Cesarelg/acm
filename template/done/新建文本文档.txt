LL ext_gcd(LL a, LL b, LL &x, LL &y) {
    if (a == 0 && b == 0) return -1;
    if (b == 0) {
        x = 1; y = 0; return a;
    }
    LL d = ext_gcd(b, a%b, y, x);
    y -= a / b * x;
    return d;
}
//ax = 1(mod n)
LL mod_reverse(LL a, LL n) {
    LL x, y, d = ext_gcd(a, n, x, y);
    if (d == 1) return (x % n + n) % n;
    else return -1;
}
